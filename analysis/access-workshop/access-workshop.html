make_graticule <- function(lats = c(-80, -50), 
                           lat_by = 10, 
                           lon_by = 45) {
  force(lats)
  make_meridians <- function(lon = 0) {
    matrix(c(lon, lon, min(lats), max(lats)), , 2, byrow = FALSE) |> 
      sf::st_linestring()
  }
  
  make_paralells <- function(lat = 0) {
    lons <- seq(0, 360, by = 1)
    lats <- rep_len(lat, length(lons))
    cbind(lons, lats) |> 
      sf::st_linestring()
  }
  
  
  c(lapply(seq(0, 360, by = lon_by), make_meridians),
    lapply(seq(min(lats), max(lats), by = lat_by), make_paralells)) |>  
    sf::st_sfc(crs = "+proj=lonlat") |> 
    sf::st_transform(crs = sic_projection)
  
}
